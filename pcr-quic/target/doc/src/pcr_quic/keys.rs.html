1//! PCR-QUIC Epoch Key Schedule
2//!
3//! This module implements the epoch key derivation for PCR-QUIC.
4//! Each epoch `e` has a shared secret `ss_e` from the hybrid KEM, from which
5//! we derive:
6//!
7//! - `PRK_epoch`: Used to derive AES encryption keys
8//! - `PRK_chain`: Used to derive IV bases for per-packet nonce ratchet
9//! - `K_server`, `K_client`: AES-256 keys for each direction
10//! - `IV_server`, `IV_client`: IV bases for nonce derivation
11//!
12//! # Key Derivation
13//!
14//! ```text
15//! ss_e (from KEM)
16//!   |
17//!   +---> PRK_epoch = HKDF(0x36 || ss, salt=epoch, info="pcr-epoch")
18//!   |       |
19//!   |       +---> K_server = HKDF(PRK_epoch, info="server aes key chain")
20//!   |       +---> K_client = HKDF(PRK_epoch, info="client aes key chain")
21//!   |
22//!   +---> PRK_chain = HKDF(0x5C || ss, salt=epoch, info="pcr-chain")
23//!           |
24//!           +---> IV_server = HKDF(PRK_chain, info="server iv key chain")
25//!           +---> IV_client = HKDF(PRK_chain, info="client iv key chain")
26//! ```
27
28use crate::pcr_shim::{self, SecretBytes};
29use crate::Result;

31/// Epoch identifier (monotonically increasing)
32pub type Epoch = u64;
33
34/// Direction of packet flow
35#[derive(Clone, Copy, Debug, PartialEq, Eq)]
36pub enum Direction {
37    /// Client sending to server
38    ClientToServer,
39    /// Server sending to client
40    ServerToClient,
41}
42
43impl Direction {
44    /// Returns the opposite direction
45    pub fn opposite(self) -> Self {
46        match self {
47            Direction::ClientToServer => Direction::ServerToClient,
48            Direction::ServerToClient => Direction::ClientToServer,
49        }
50    }
51}

53/// Keys derived for a single epoch
54///
55/// Contains all cryptographic material needed for packet protection
56/// within an epoch. Should be securely zeroized when the epoch is retired.
57#[derive(Clone)]
58pub struct EpochKeys {
59    /// The epoch number these keys belong to
60    pub epoch: Epoch,
61    
62    /// AES-256 key for sending packets
63    pub k_send: SecretBytes<32>,
64    
65    /// AES-256 key for receiving packets
66    pub k_recv: SecretBytes<32>,
67    
68    /// IV base for sending (used with per-packet nonce ratchet)
69    /// Only first 12 bytes are used as the IV
70    pub iv_send: SecretBytes<32>,
71    
72    /// IV base for receiving (used with per-packet nonce ratchet)
73    /// Only first 12 bytes are used as the IV
74    pub iv_recv: SecretBytes<32>,
75    
76    /// PRK for epoch key derivation (kept for potential rekeying)
77    pub prk_epoch: SecretBytes<32>,
78    
79    /// PRK for chain/IV derivation (used by per-packet ratchet)
80    pub prk_chain: SecretBytes<32>,
81}
82
83impl EpochKeys {
84    /// Get the 12-byte IV for sending
85    pub fn iv_send_12(&self) -> &[u8; 12] {
86        self.iv_send.as_bytes()[..12].try_into().unwrap()
87    }
88    
89    /// Get the 12-byte IV for receiving
90    pub fn iv_recv_12(&self) -> &[u8; 12] {
91        self.iv_recv.as_bytes()[..12].try_into().unwrap()
92    }
93}

95/// HKDF prefix byte for epoch PRK derivation (HMAC outer pad XOR pattern)
96const PRK_EPOCH_PREFIX: u8 = 0x36;
97
98/// HKDF prefix byte for chain PRK derivation (HMAC inner pad XOR pattern)
99const PRK_CHAIN_PREFIX: u8 = 0x5C;
100
101/// Info string for epoch PRK derivation
102const INFO_PCR_EPOCH: &[u8] = b"pcr-epoch";
103
104/// Info string for chain PRK derivation
105const INFO_PCR_CHAIN: &[u8] = b"pcr-chain";
106
107/// Info string for server AES key derivation
108const INFO_SERVER_AES_KEY: &[u8] = b"server aes key chain";
109
110/// Info string for client AES key derivation
111const INFO_CLIENT_AES_KEY: &[u8] = b"client aes key chain";
112
113/// Info string for server IV derivation
114const INFO_SERVER_IV: &[u8] = b"server iv key chain";
115
116/// Info string for client IV derivation
117const INFO_CLIENT_IV: &[u8] = b"client iv key chain";
118
119/// Derive PRK_epoch and PRK_chain from a shared secret
120///
121/// # Arguments
122/// * `ss` - 32-byte shared secret from KEM
123/// * `epoch` - Current epoch number (used as salt)
124///
125/// # Returns
126/// Tuple of (PRK_epoch, PRK_chain), each 32 bytes
127///
128/// # Security
129/// The shared secret `ss` should be zeroized immediately after this call.
130pub fn derive_prks(ss: &[u8; 32], epoch: Epoch) -> Result<(SecretBytes<32>, SecretBytes<32>)> {
131    // Build IKM for PRK_epoch: 0x36 || ss
132    let mut ikm_epoch = Vec::with_capacity(1 + ss.len());
133    ikm_epoch.push(PRK_EPOCH_PREFIX);
134    ikm_epoch.extend_from_slice(ss);
135
136    // Build IKM for PRK_chain: 0x5C || ss
137    let mut ikm_chain = Vec::with_capacity(1 + ss.len());
138    ikm_chain.push(PRK_CHAIN_PREFIX);
139    ikm_chain.extend_from_slice(ss);
140
141    // Use epoch as salt (big-endian)
142    let salt = epoch.to_be_bytes();
143
144    // Derive PRKs
145    let prk_epoch_vec = pcr_shim::hkdf_sha256(&ikm_epoch, &salt, INFO_PCR_EPOCH, 32)?;
146    let prk_chain_vec = pcr_shim::hkdf_sha256(&ikm_chain, &salt, INFO_PCR_CHAIN, 32)?;
147
148    // Convert to fixed-size arrays with secure storage
149    let mut prk_epoch = SecretBytes::<32>::new();
150    prk_epoch.as_bytes_mut().copy_from_slice(&prk_epoch_vec);
151
152    let mut prk_chain = SecretBytes::<32>::new();
153    prk_chain.as_bytes_mut().copy_from_slice(&prk_chain_vec);
154
155    // Zeroize temporary vectors
156    let mut ikm_epoch = ikm_epoch;
157    let mut ikm_chain = ikm_chain;
158    pcr_shim::secure_zero(&mut ikm_epoch);
159    pcr_shim::secure_zero(&mut ikm_chain);
160
161    Ok((prk_epoch, prk_chain))
162}

164/// Derive all epoch keys from a shared secret
165///
166/// # Arguments
167/// * `ss` - 32-byte shared secret from KEM (will NOT be zeroized - caller must do this)
168/// * `epoch` - Current epoch number
169/// * `is_client` - True if we are the client, false if server
170///
171/// # Returns
172/// `EpochKeys` containing all derived keys for this epoch
173///
174/// # Security
175/// - The shared secret `ss` should be zeroized by the caller after this returns
176/// - The returned `EpochKeys` should be zeroized when the epoch is retired
177pub fn derive_epoch_keys(ss: &[u8; 32], epoch: Epoch, is_client: bool) -> Result<EpochKeys> {
178    // Derive PRKs from shared secret
179    let (prk_epoch, prk_chain) = derive_prks(ss, epoch)?;
180
181    // Derive AES keys from PRK_epoch
182    let k_server_vec = pcr_shim::hkdf_sha256(
183        prk_epoch.as_bytes(),
184        &[],
185        INFO_SERVER_AES_KEY,
186        32,
187    )?;
188    let k_client_vec = pcr_shim::hkdf_sha256(
189        prk_epoch.as_bytes(),
190        &[],
191        INFO_CLIENT_AES_KEY,
192        32,
193    )?;
194
195    // Derive IVs from PRK_chain
196    let iv_server_vec = pcr_shim::hkdf_sha256(
197        prk_chain.as_bytes(),
198        &[],
199        INFO_SERVER_IV,
200        32,
201    )?;
202    let iv_client_vec = pcr_shim::hkdf_sha256(
203        prk_chain.as_bytes(),
204        &[],
205        INFO_CLIENT_IV,
206        32,
207    )?;
208
209    // Convert to SecretBytes
210    let mut k_server = SecretBytes::<32>::new();
211    k_server.as_bytes_mut().copy_from_slice(&k_server_vec);
212
213    let mut k_client = SecretBytes::<32>::new();
214    k_client.as_bytes_mut().copy_from_slice(&k_client_vec);
215
216    let mut iv_server = SecretBytes::<32>::new();
217    iv_server.as_bytes_mut().copy_from_slice(&iv_server_vec);
218
219    let mut iv_client = SecretBytes::<32>::new();
220    iv_client.as_bytes_mut().copy_from_slice(&iv_client_vec);
221
222    // Assign send/recv based on role
223    let (k_send, k_recv, iv_send, iv_recv) = if is_client {
224        (k_client, k_server, iv_client, iv_server)
225    } else {
226        (k_server, k_client, iv_server, iv_client)
227    };
228
229    Ok(EpochKeys {
230        epoch,
231        k_send,
232        k_recv,
233        iv_send,
234        iv_recv,
235        prk_epoch,
236        prk_chain,
237    })
238}

240/// Derive epoch keys for a specific direction only
241///
242/// This is useful when you only need keys for one direction (e.g., during
243/// initial handshake when only one party is sending).
244///
245/// # Arguments
246/// * `ss` - 32-byte shared secret from KEM
247/// * `epoch` - Current epoch number
248/// * `direction` - Which direction to derive keys for
249///
250/// # Returns
251/// Tuple of (AES key, IV base) for the specified direction
252pub fn derive_direction_keys(
253    ss: &[u8; 32],
254    epoch: Epoch,
255    direction: Direction,
256) -> Result<(SecretBytes<32>, SecretBytes<32>)> {
257    let (prk_epoch, prk_chain) = derive_prks(ss, epoch)?;
258
259    let (key_info, iv_info) = match direction {
260        Direction::ClientToServer => (INFO_CLIENT_AES_KEY, INFO_CLIENT_IV),
261        Direction::ServerToClient => (INFO_SERVER_AES_KEY, INFO_SERVER_IV),
262    };
263
264    let key_vec = pcr_shim::hkdf_sha256(prk_epoch.as_bytes(), &[], key_info, 32)?;
265    let iv_vec = pcr_shim::hkdf_sha256(prk_chain.as_bytes(), &[], iv_info, 32)?;
266
267    let mut key = SecretBytes::<32>::new();
268    key.as_bytes_mut().copy_from_slice(&key_vec);
269
270    let mut iv = SecretBytes::<32>::new();
271    iv.as_bytes_mut().copy_from_slice(&iv_vec);
272
273    Ok((key, iv))
274}

280    #[test]
281    fn test_derive_prks() {
282        let ss = [0x42u8; 32];
283        let epoch = 1;
284
285        let (prk_epoch, prk_chain) = derive_prks(&ss, epoch).unwrap();
286
287        // PRKs should be different
288        assert_ne!(prk_epoch.as_bytes(), prk_chain.as_bytes());
289
290        // PRKs should be deterministic
291        let (prk_epoch2, prk_chain2) = derive_prks(&ss, epoch).unwrap();
292        assert_eq!(prk_epoch.as_bytes(), prk_epoch2.as_bytes());
293        assert_eq!(prk_chain.as_bytes(), prk_chain2.as_bytes());
294
295        // Different epoch should give different PRKs
296        let (prk_epoch3, prk_chain3) = derive_prks(&ss, epoch + 1).unwrap();
297        assert_ne!(prk_epoch.as_bytes(), prk_epoch3.as_bytes());
298        assert_ne!(prk_chain.as_bytes(), prk_chain3.as_bytes());
299    }

302    fn test_derive_epoch_keys_client_server_symmetry() {
303        let ss = [0x42u8; 32];
304        let epoch = 1;
305
306        let client_keys = derive_epoch_keys(&ss, epoch, true).unwrap();
307        let server_keys = derive_epoch_keys(&ss, epoch, false).unwrap();
308
309        // Client's send key should be server's recv key

311        assert_eq!(client_keys.k_recv.as_bytes(), server_keys.k_send.as_bytes());
312
313        // Same for IVs
314        assert_eq!(client_keys.iv_send.as_bytes(), server_keys.iv_recv.as_bytes());
315        assert_eq!(client_keys.iv_recv.as_bytes(), server_keys.iv_send.as_bytes());
316
317        // PRKs should be the same
318        assert_eq!(client_keys.prk_epoch.as_bytes(), server_keys.prk_epoch.as_bytes());
319        assert_eq!(client_keys.prk_chain.as_bytes(), server_keys.prk_chain.as_bytes());
320    }
