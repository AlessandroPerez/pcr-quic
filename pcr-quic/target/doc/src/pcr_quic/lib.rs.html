1//! # PCR-QUIC: Double Ratchet Protocol for QUIC
2//!
3//! This crate implements a double ratchet protocol providing:
4//! - **Forward Secrecy (FS)**: Per-packet nonce ratchet
5//! - **Post-Compromise Security (PCS)**: KEM-based epoch rekey
6//! - **Precise Key Deletion**: Cryptographic key zeroization
7//!
8//! ## Architecture
9//!
10//! PCR-QUIC operates with two ratchet layers:
11//!
12//! 1. **Epoch Ratchet**: KEM-based rekey (~2 minutes)
13//!    - Hybrid X25519 + ML-KEM-768 key exchange
14//!    - Derives per-direction AES-256-GCM keys
15//!
16//! 2. **Per-Packet Ratchet**: BLAKE3-based nonce derivation
17//!    - Computes nonce from packet number and epoch state
18//!    - Forward-secure nonce key deletion

22// ============================================================================
23// Error Types (defined first so From impls are available to all modules)

26/// PCR-QUIC error types
27#[derive(Debug, Clone, PartialEq, Eq)]
28pub enum PcrError {
29    /// Cryptographic operation failed
30    CryptoFailed,
31    
32    /// Buffer too small for operation
33    BufferTooShort,
34    
35    /// Invalid packet number
36    InvalidPacketNumber,
37    
38    /// Epoch not found
39    EpochNotFound,
40    
41    /// Invalid frame format
42    InvalidFrame,
43    
44    /// Done processing (not an error)
45    Done,
46    
47    /// Invalid state transition
48    InvalidState,
49    
50    /// KEM operation failed
51    KemFailed,
52    
53    /// HKDF operation failed
54    HkdfFailed,
55    
56    /// AEAD operation failed
57    AeadFailed,
58    
59    /// Packet decryption failed (authentication)
60    DecryptionFailed,
61    
62    /// Generic error with message
63    Other(String),
64}
65
66impl fmt::Display for PcrError {
67    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
68        match self {
69            PcrError::CryptoFailed => write!(f, "Cryptographic operation failed"),
70            PcrError::BufferTooShort => write!(f, "Buffer too short"),
71            PcrError::InvalidPacketNumber => write!(f, "Invalid packet number"),
72            PcrError::EpochNotFound => write!(f, "Epoch not found"),
73            PcrError::InvalidFrame => write!(f, "Invalid frame format"),
74            PcrError::Done => write!(f, "Done"),
75            PcrError::InvalidState => write!(f, "Invalid state"),
76            PcrError::KemFailed => write!(f, "KEM operation failed"),
77            PcrError::HkdfFailed => write!(f, "HKDF operation failed"),
78            PcrError::AeadFailed => write!(f, "AEAD operation failed"),
79            PcrError::DecryptionFailed => write!(f, "Packet decryption failed"),
80            PcrError::Other(msg) => write!(f, "{}", msg),
81        }
82    }
83}
84
85impl std::error::Error for PcrError {}
86
87/// Result type for PCR-QUIC operations
88pub type Result<T> = std::result::Result<T, PcrError>;
